    –ü–æ–ª—É—á–∏—Ç—å –æ—Å—Ç–∞—Ç–∫–∏ FBO —á–µ—Ä–µ–∑ Statistics API

def _get_fbo_stocks_statistics(self) -> List[Dict]:
    """
    –í–Ω—É—Ç—Ä–µ–Ω–Ω–∏–π –º–µ—Ç–æ–¥: Statistics API
    """
    all_stocks = []
    date_from = "2019-06-20"  # –°–∞–º–∞—è —Ä–∞–Ω–Ω—è—è –¥–∞—Ç–∞ –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –≤—Å–µ—Ö –¥–∞–Ω–Ω—ã—Ö
    
    while True:
        response = self.api.get(
            "/api/v1/supplier/stocks",
            params={"dateFrom": date_from},
            base_url=self.api.config.statistics_url  # –î–æ–±–∞–≤–∏—Ç—å –≤ WBConfig!
        )
        
        if not isinstance(response, list):
            break
            
        if not response:
            break
        
        # –§–∏–ª—å—Ç—Ä—É–µ–º —Ç–æ–ª—å–∫–æ FBO —Å–∫–ª–∞–¥—ã (—Å–∫–ª–∞–¥—ã WB)
        fbo_stocks = [
            stock for stock in response 
            if self._is_fbo_warehouse(stock.get('warehouseName', ''))
        ]
        
        all_stocks.extend(fbo_stocks)
        
        # –ü–∞–≥–∏–Ω–∞—Ü–∏—è —á–µ—Ä–µ–∑ lastChangeDate
        last_change = response[-1].get('lastChangeDate')
        if not last_change or last_change == date_from:
            break
            
        date_from = last_change
        print(f"DEBUG: –ü–∞–≥–∏–Ω–∞—Ü–∏—è, —Å–ª–µ–¥—É—é—â–∞—è –¥–∞—Ç–∞: {date_from}")
    
    return all_stocks
